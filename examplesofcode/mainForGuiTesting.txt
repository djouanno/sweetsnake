    public static void main(final String[] args) {
        final Gui gui = new Gui();
        gui.init();
        final GameView gv = new GameView();
        gv.gui = gui;

        final ComponentDTO[][] elems = new ComponentDTO[20][20];
        elems[0][0] = new ComponentDTO("id1", 0, 0, ComponentType.SNAKE);
        elems[19][19] = new ComponentDTO("id2", 19, 19, ComponentType.SNAKE);
        final GameBoardDTO dto = new GameBoardDTO(20, 20, elems, 10);

        gv.setGameboardDto(dto);
        gv.init();
        gv.build();
        gui.getContentPane().removeAll();
        gui.getContentPane().add(gv);
        gui.refreshUI();
    }
